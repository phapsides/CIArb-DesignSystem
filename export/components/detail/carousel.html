export default class Carousel {
	constructor($module) {

		this.$slider.setAttribute("id", id);
		const slider = this.$slider;

		let $breakpoints = {};
		if (!$hero) {
			$breakpoints = {
				768: {

					slidesPerGroup: 3,
					spaceBetween: 30
